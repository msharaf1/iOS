struct Rectangle {
    var width = 10
    var height = 20

    func printDesc(){

    }
    mutating func doubleWidth(){
        width *= 2;
    }
}

//print(myRectangle.height)
myRectangle.printDesc()
print(myRectangle.doubleWidth())
